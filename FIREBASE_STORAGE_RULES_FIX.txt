If you want to restrict uploads to admins only (recommended for production):

rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {
    // Founder images - only admins can upload, anyone can read
    match /founders/{imageId} {
      allow read: if true;
      allow write: if request.auth != null && 
                     request.auth.token.role == 'admin';
    }
    
    // Default rules for other paths
    match /{allPaths=**} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}

Note: This requires your user to have role: 'admin' in their custom claims.
